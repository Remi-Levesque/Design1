\begin{enumerate}
   \item \textbf{Ouverture de la réunion}
   \item \textbf{Nomination ou confirmation du président et du secrétaire}
   \item \textbf{Lecture et adoption de l'ordre du jour}
   \item \textbf{Lecture et adoption du procès-verbal de la réunion du 7 février 2019}
   \item \textbf{Affaires découlant du procès-verbal}
      \begin{enumerate}
         \item Investigations sur la caméra à utiliser
            \begin{enumerate}
               \item Considérations nocturnes
               \item Collecte de données et compatibilité avec le logiciel
               \item Coûts
               \item Faisabilité du logiciel
               \item Processus d'apprentissage et vision numérique
            \end{enumerate}
	\item Retour sur le commencement du chapitre 3
	\begin{enumerate}
		\item Besoins
		\item Objectifs
	\end{enumerate}
      \end{enumerate}
   \item \textbf{Points à traiter}
      \begin{enumerate}
         \item Aspects mécaniques: matériaux de conception et durabilité
            \begin{enumerate}
               \item Valider les contraintes maximales de la caméra ainsi que sa tenue à l'oxydation
               \item Durée de vie et température
            \end{enumerate}
         \item Aspects de sécurité: création d'un réseau de transfert de données efficace
         \begin{enumerate}
               \item Types de transfert de données possibles: évaluer les cas de perte de signal
               ou de vol.
               \item Adapter le signal à un milieu marin. (Quel type de signal serait le plus fiable sous l'eau et à distance?)
            \end{enumerate}
      \end{enumerate}
   \item \textbf{Divers}
   \item \textbf{Répartition des tâches}
   \item \textbf{Évaluation de la réunion}
   \item \textbf{Date, heure, lieu et objectif de la prochaine réunion}
   \item \textbf{Fermeture de la réunion}
\end{enumerate}
